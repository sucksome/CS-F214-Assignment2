107int getPostfix(char* expression, char * stack, int lenInfix)
108{
109 int i, j;
110
111 for (i = 0, j = -1; expression[i]; ++i)
112 {
113 // we check whether the scanned character is an operand or not and add this to the output
114 if (checkIfOperand(expression[i])) expression[++j] = expression[i];
115
116 // we push '(' to the stack if we scan it
117 else if (expression[i] == '(') push(expression[i],lenInfix,stack);
118
119 // if the scanned character is ')', we need to pop from the stack till '(' is encountered in the stack
120 else if (expression[i] == ')')
121 {
122 while (!isEmpty(stack) && peek(stack) != '(') expression[++j] = pop(stack);

125 else pop(stack);
126 }

133 }

136 while (!isEmpty(stack)) expression[++j] = pop(stack);
